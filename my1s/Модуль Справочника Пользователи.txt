
Процедура ПередЗаписью(Отказ)
	ТС = ЭтотОбъект.Ссылка;
	ТЛ = ЭтотОбъект.Наименование;
	
	Запрос = Новый Запрос;
	Запрос.Текст =
	"ВЫБРАТЬ
	|	Ссылка
	|ИЗ
	|	Справочник.Пользователи
	|ГДЕ
	|	Наименование = &Логин
	|	И Ссылка <> &ТекСсылка";
	
	Запрос.УстановитьПараметр("Логин", ТЛ);
	Запрос.УстановитьПараметр("ТекСсылка", ТС);
	
	Выборка = Запрос.Выполнить().Выбрать();
	Если Выборка.Количество()>0 Тогда
		Отказ = Истина;
		Сообщить("не уникальный логин")
	КонецЕсли;
КонецПроцедуры
