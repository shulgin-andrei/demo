&НаСервере
Функция ИзменитьПарольНаСервере()
	Если Параметры.ПараметрП.Пароль = ТекущийПароль И НовыйПароль = Подтверждение Тогда
		ПольО = Параметры.ПараметрП.ПолучитьОбъект();
		ПольО.Пароль = НовыйПароль;
		ПольО.ДатаВхода = ТекущаяДата();
		ПольО.Записать();
		Возврат Истина;
	Иначе
		Сообщить("пароли не совпадают");
		Возврат Ложь;
	КонецЕсли;		
КонецФункции

&НаКлиенте
Процедура ИзменитьПароль(Команда)
	Если ИзменитьПарольНаСервере() = Истина Тогда
		Если Роль = ПредопределенноеЗначение("Перечисление.Роли.Пользователь") Тогда
			Сообщить("успешн смен пароль");
			ОткрытьФорму("ОбщаяФорма.ФормаПользователя");
			ЭтаФорма.Закрыть();
		КонецЕсли; 
		Если Роль = ПредопределенноеЗначение("Перечисление.Роли.Админ") Тогда
			Сообщить("успешн смен пароль");
			ОткрытьФорму("Справочник.Пользователи.ФормаСписка");
			ЭтаФорма.Закрыть();
		КонецЕсли;
	КонецЕсли;
КонецПроцедуры

&НаСервере
Процедура ПриОткрытииНаСервере()
	Роль = Параметры.ПараметрП.Роль;

Процедура ПриОткрытии(Отказ)
	ПриОткрытииНаСервере();
