&НаСервере
Функция ВойтиНаСервере()
	ПольН = Справочники.Пользователи.ПустаяСсылка();
	Если НЕ Справочники.Пользователи.НайтиПоНаименованию(Логин, Истина).Пустая() Тогда
		ПольН = Справочники.Пользователи.НайтиПоНаименованию(Логин, Истина).Ссылка;
		Если ПольН.Пароль = Пароль Тогда
			Если НЕ ПольН.ДатаВхода = Дата("01.01.0001 0:00:00") Тогда
				Если ТекущаяДата()> ДобавитьМесяц(НачалоМесяца(ПольН.ДатаВхода),1) Тогда
					УстРекв(ПольН, "Активность", Ложь);
					Сообщить("по времни истек акк");
				КонецЕсли;
			КонецЕсли;
			Возврат ПольН.Роль;
		Иначе
			УстПопытки(ПольН);
			Сообщить("пароль не тот");
			Возврат Неопределено;
		КонецЕсли;
	Иначе
		Сообщить("логин не тот");
		Возврат Неопределено;
	КонецЕсли;
КонецФункции

&НаКлиенте
Процедура Войти(Команда)
	Роль = ВойтиНаСервере(); 
	Если ЗначениеЗаполнено(Роль) Тогда
		Если АктивностьПоль(ПольН) = Истина Тогда 
			УстРекв(ПольН,"КолПопыток",0);
			Если ВходПоль(ПольН) = Дата("01.01.0001 0:00:00") Тогда
				Сообщить("смените пароль");
				ОткрытьФорму("ОбщаяФорма.ФормаСменыПароля", Новый Структура("ПараметрП", ПольН));
				ЭтаФорма.Закрыть();
			Иначе 
				УстРекв(ПольН,"ДатаВхода",ТекущаяДата());
				Если Роль = ПредопределенноеЗначение("Перечисление.Роли.Пользователь") Тогда
					Сообщить("найс вход");
					ОткрытьФорму("ОбщаяФорма.ФормаПользователя");
				КонецЕсли;
				Если Роль = ПредопределенноеЗначение("Перечисление.Роли.Админ") Тогда
					Сообщить("найс вход");
					ОткрытьФорму("Справочник.Пользователи.ФормаСписка");
				КонецЕсли;
				ЭтаФорма.Закрыть();
			КонецЕсли;			
		Иначе
			УстРекв(ПольН,"Активность",Ложь);
			Сообщить("бан");
		КонецЕсли; 
	Иначе
		Если ПопыткиПоль(ПольН)> 3 Тогда 
			УстРекв(ПольН,"Активность",Ложь);
			Сообщить("бан по попыткам");  
		КонецЕсли;
	КонецЕсли;
КонецПроцедуры

Процедура УстПопытки(Поль)
	ПольО = Поль.ПолучитьОбъект();
	ПольО.КолПопыток = ПольО.КолПопыток+1;
	ПольО.Записать();
